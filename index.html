<script>
const READINGS = ["Бытие 1-3","Бытие 4:1-6:9","Бытие 6:9-9:29","Бытие 10,11","Бытие 12-14","Бытие 15-17","Бытие 18,19","Бытие 20-22","Бытие 23,24","Б[...]
const OT_END_INDEX = 272;
const STORAGE_KEY = 'bible-plan-v2';

let checkedState = {};

const listEl = document.getElementById('list');
const progressFill = document.getElementById('progressFill');
const progressLabel = document.getElementById('progressLabel');

function buildList() {
  const frag = document.createDocumentFragment();
  const ot = document.createElement('div');
  ot.innerHTML = '<div style="text-align:center;color:#1a4d8f;font-style:italic;margin:20px 0 10px 0;">Еврейские Писания</div>';
  frag.appendChild(ot);

  READINGS.forEach((text, i) => {
    if (i === OT_END_INDEX) {
      const nt = document.createElement('div');
      nt.innerHTML = '<div style="text-align:center;color:#1a4d8f;font-style:italic;margin:20px 0 10px 0;">Христианские Греческие Писания</div>';
      frag.appendChild(nt);
    }
    const li = document.createElement('li');
    li.className = 'row';
    li.dataset.index = i;
    li.innerHTML = `
      <div class="box"></div>
      <div class="num">${i+1}</div>
      <div class="txt">${text}</div>
    `;
    li.addEventListener('click', () => toggle(i));
    frag.appendChild(li);
  });
  listEl.appendChild(frag);
}

function toggle(i) {
  checkedState[i] = !checkedState[i];
  render();
  save();
}

function render() {
  let count = 0;
  document.querySelectorAll('li.row').forEach(row => {
    const idx = parseInt(row.dataset.index);
    const checked = !!checkedState[idx];
    row.classList.toggle('checked', checked);
    if (checked) count++;
  });
  const perc = (count / READINGS.length * 100).toFixed(1);
  progressFill.style.width = perc + '%';
  progressLabel.textContent = count + ' / ' + READINGS.length;
}

function save() {
  localStorage.setItem(STORAGE_KEY, JSON.stringify(checkedState));
}

function load() {
  const saved = localStorage.getItem(STORAGE_KEY);
  if (saved) checkedState = JSON.parse(saved);
  render();
}

buildList();
load();

// Register service worker for PWA support
if ('serviceWorker' in navigator) {
  navigator.serviceWorker.register('/Bible-reading/sw.js').catch(() => {});
}
</script>
